### Quick start

```swift showLineNumbers copy
import Graphiti
import Pioneer

struct Resolver { ... }

let schema = try Schema<Resolver, Void> { ... }

let server = Pioneer(
    schema: schema,
    resolver: .init()
)

try server.standaloneServer(
    port: 4000,
    host: "127.0.0.1"
)
```
